233 template < typename ConnectionT >
234 nest::ConnectorBase* DiligentConnectorModel< ConnectionT >::get_hom_connector( nest::ConnectorBase* conn,
235  nest::synindex syn_id )
236 {
237  if (!conn)
238  {
239  return 0;
240  }
241  else if (conn->homogeneous_model())
242  {
243  if (conn->get_syn_id() == syn_id)
244  return conn;
245  }
246  else
247  {
248  // connector is heterogeneous - go through all entries and search for correct syn_id
249  nest::HetConnector* hc = static_cast< nest::HetConnector* >( conn );
250  for ( size_t i = 0; i < hc->size(); i++ )
251  {
252  // need to cast to vector_like to access syn_id
253  if ( ( *hc )[ i ]->get_syn_id() == syn_id ) // find entry for this type
254  {
255  return ( *hc )[ i ];
256  }
257  }
258  }
259 
260  return 0;
261 }
